import telebot
import yt_dlp
import subprocess
import os
import tempfile
from telebot import types

TOKEN = os.getenv("BOT_TOKEN")
if not TOKEN:
    TOKEN = "8754083614:AAEgAhfkJIimlqrfJ03O0solwS12Hcl74Oo"  # solo para pruebas locales

bot = telebot.TeleBot(TOKEN)

@bot.message_handler(func=lambda message: True)
def procesar_reel(message):
    url = message.text.strip()
    
    if not any(domain in url.lower() for domain in ["instagram.com", "tiktok.com", "facebook.com/reels", "youtube.com/shorts", "youtu.be"]):
        bot.reply_to(message, "❌ Por favor, enviame solo el link del Reel, TikTok o Short.")
        return

    processing_msg = bot.reply_to(message, "🔄 Descargando el video bro...\nEsto puede tardar un rato, relajá.")

    try:
        with tempfile.TemporaryDirectory() as tmpdir:
            output_template = f'{tmpdir}/original.%(ext)s'
            
            ydl_opts = {
                'outtmpl': output_template,
                'quiet': True,
                'no_warnings': True,
                'format': 'bestvideo+bestaudio/best',
                'merge_output_format': 'mp4',
                'keepvideo': False,
            }

            with yt_dlp.YoutubeDL(ydl_opts) as ydl:
                info = ydl.extract_info(url, download=True)
                video_path = ydl.prepare_filename(info)

            # Aplicar efecto espejo (flip horizontal)
            final_video = f'{tmpdir}/mirror_final.mp4'
            subprocess.run([
                'ffmpeg', '-i', video_path,
                '-vf', 'hflip',                    # ← espejo horizontal
                '-c:v', 'libx264', '-preset', 'medium', '-crf', '23',
                '-c:a', 'aac', '-b:a', '128k',
                '-movflags', '+faststart',
                final_video
            ], check=True, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)

            # Enviar el video procesado
            with open(final_video, 'rb') as video_file:
                bot.send_video(
                    chat_id=message.chat.id,
                    video=video_file,
                    caption="✅ ¡Listo!\n🔄 Video volteado en espejo + sin marcas de agua\n\nviola mentes tranquilo ahora 🔥",
                    reply_to_message_id=message.message_id,
                    supports_streaming=True
                )

            bot.delete_message(message.chat.id, processing_msg.message_id)

    except Exception as e:
        error_text = str(e)
        if "Sign in" in error_text or "login" in error_text.lower():
            bot.edit_message_text("⚠️ Este contenido requiere login (probablemente Instagram privado). Por ahora solo funciona con reels públicos.", message.chat.id, processing_msg.message_id)
        else:
            bot.edit_message_text(f"❌ manco le pifiaste...\nIntenta con otro video o enviame el error completo.", message.chat.id, processing_msg.message_id)

print("🚀 Bot de Reels Mirror iniciado correctamente...")
bot.infinity_polling()
